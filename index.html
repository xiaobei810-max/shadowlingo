const PRAISES = [

  { min:90, msgs:["🏆 完美！声调拿捏精准！","🌟 太棒了，说得非常地道！"] },

  { min:75, msgs:["🔥 非常好，再练一遍就完美！","💪 发音很稳，越来越自然了！"] },

  { min:60, msgs:["👏 不错，节奏感已经到位了！","💡 有进步，多听几遍能做到！"] },

let idx = 0;

let phase = "idle";

let segTimer = null;

let cdTimer = null;

let hasRecorded = false;


const audio = new Audio(AUDIO_SRC);

audio.preload = "auto";


function randScore() { return Math.floor(Math.random() * 14) + 85; }

function getPraise(s) {

  const b = PRAISES.find(p => s >= p.min);

  return b.msgs[Math.floor(Math.random() * b.msgs.length)];

}


function speak(on) {

  const ring = document.getElementById("ring");

  const mascot = document.getElementById("mascot");

  if (on) {

    ring.style.opacity = "1";

    ring.style.animation = "glow 1.2s ease-in-out infinite";

    mascot.style.animation = "pulse 0.4s ease-in-out infinite alternate";

  } else {

    ring.style.opacity = "0";

    ring.style.animation = "";

    mascot.style.animation = "";

  }

}


function buildKara(text) {

  const el = document.getElementById("kara");

  el.innerHTML = "";

  if (!text) return;

  [...text].forEach(c => {

    const span = document.createElement("span");

    span.className = "kc";

    span.textContent = c;

    el.appendChild(span);

  });

}


function completeKara() {

  document.querySelectorAll(".kc").forEach(c => {

    c.classList.remove("active");

    c.classList.add("done");

  });

}


function setStatus(txt) {

  document.getElementById("status").textContent = txt;

}


function updateUI() {

  const s = SENTENCES[idx];

  document.getElementById("counter").textContent = (idx+1) + " / " + SENTENCES.length;

  document.getElementById("lbl").textContent = "第 " + (idx+1) + "/" + SENTENCES.length + " 句";

  document.getElementById("prog").style.width = ((idx / SENTENCES.length) * 100) + "%";

  document.getElementById("pinyin").textContent = s.pinyin;

  document.getElementById("eng").textContent = s.en;


  const bl = document.getElementById("bl");

  const bs = document.getElementById("bs");

  const bn = document.getElementById("bn");

  const bdone = document.getElementById("bdone");

  const bback = document.getElementById("bback");


  bs.disabled = (phase === "idle");

  bn.disabled = (phase === "idle" || phase === "listening");

  bback.disabled = (idx === 0);


  bl.className = "btn" + (phase === "listening" ? " active" : "");

  bs.className = "btn" + (phase === "slow" ? " active" : "");

  bn.className = "btn" + (phase === "normal" ? " active" : "");

  bdone.className = "btn btn-done" + (phase === "scored" ? " show" : "");


  const tabs = ["t0","t1","t2","t3"];

  const tabPhase = {idle:0, listening:0, slow:1, normal:2, scored:3};

  const cur = tabPhase[phase] || 0;

  tabs.forEach((id,i) => {

    document.getElementById(id).className = "tab" + (i === cur ? " active" : "");

  });


  buildKara(s.zh);

  document.getElementById("score-card").classList.remove("show");

}


function playSegment(start, end, rate, onEnd) {

  clearTimeout(segTimer);

  audio.playbackRate = rate;

  audio.currentTime = start;

  audio.play().catch(() => {});

  speak(true);

  const dur = ((end - start) / rate) * 1000;

  const fill = document.getElementById("bar-fill");

  segTimer = setTimeout(function() {

    audio.pause();

    speak(false);

    completeKara();

    if (onEnd) onEnd();

  }, dur + 100);

}


async function startRec(dur, onDone) {

  hasRecorded = false;

  try {

    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });

    hasRecorded = true;

    const mr = new MediaRecorder(stream);

    mr.start();

    setTimeout(function() {

      mr.stop();

      stream.getTracks().forEach(t => t.stop());

      onDone();

    }, dur);

  } catch(e) {

    hasRecorded = false;

    showNoMic();

  }

}


function showNoMic() {

  phase = "scored";

  updateUI();

  document.getElementById("praise").textContent = "🎙 需要麦克风权限 Mic needed";

  document.getElementById("hint").textContent = "请在浏览器地址栏允许麦克风 / Allow mic in browser settings";

  document.getElementById("sn").textContent = "-";

  document.getElementById("sc").setAttribute("stroke-dasharray", "0 188");

  document.getElementById("score-card").classList.add("show");

}


function showScore() {

  if (!hasRecorded) { showNoMic(); return; }

  phase = "scored";

  updateUI();

  const s = randScore();

  const fb = FEEDBACK[idx] || FEEDBACK[0];

  const circ = 2 * Math.PI * 30;

  const dash = (s / 100) * circ;

  const color = s >= 85 ? "#22c55e" : s >= 70 ? "#facc15" : "#f87171";

  const sc = document.getElementById("sc");

  sc.setAttribute("stroke", color);

  setTimeout(function() {

    sc.setAttribute("stroke-dasharray", dash + " " + circ);

  }, 100);

  document.getElementById("sn").textContent = s;

  document.getElementById("sn").style.color = color;

  document.getElementById("praise").textContent = fb.praise + "  " + getPraise(s);

  document.getElementById("hint").textContent = fb.hint;

  document.getElementById("score-card").classList.add("show");

}


function doListen() {

  clearTimeout(segTimer); audio.pause(); speak(false);

  phase = "listening"; setStatus("🎧 播放中..."); updateUI();

  const s = SENTENCES[idx];

  playSegment(s.start, s.end, 1, function() {

    phase = "listened"; setStatus("准备好了"); updateUI();

  });

}


function doSlow() {

  if (phase === "idle") return;

  clearTimeout(segTimer); audio.pause(); speak(false);

  phase = "slow"; setStatus("🐢 慢速跟读..."); updateUI();

  const s = SENTENCES[idx];

  const dur = ((s.end - s.start) / 0.65) * 1000;

  playSegment(s.start, s.end, 0.65, function() { setStatus("🎙 录音中..."); });

  startRec(dur + 500, function() {

    phase = "slowed"; setStatus("录音完成"); showScore();

  });

}


function doNormal() {

  if (phase === "idle" || phase === "listening") return;

  clearTimeout(segTimer); audio.pause(); speak(false);

  phase = "normal"; setStatus("🚀 常速跟读..."); updateUI();

  const s = SENTENCES[idx];

  const dur = (s.end - s.start) * 1000;

  playSegment(s.start, s.end, 1, function() { setStatus("🎙 录音中..."); });

  startRec(dur + 500, function() {

    phase = "normal_done"; setStatus("录音完成"); showScore();

  });

}


function doBack() {

  if (idx === 0) return;

  clearTimeout(segTimer); audio.pause(); speak(false);

  idx--;

  phase = "idle";

  setStatus("准备好了");

  updateUI();

}


function doNext() {

  clearTimeout(segTimer); audio.pause(); speak(false);

  if (idx >= SENTENCES.length - 1) {

    alert("🎉 恭喜完成本课！Congratulations!");

    return;

  }

  idx++;

  phase = "idle";

  setStatus("准备好了");

  updateUI();

}


// Karaoke timeupdate sync

const RATE_OFFSET = { fast: 0.15, slow: 0.3 };

audio.addEventListener("timeupdate", function() {

  const s = SENTENCES[idx];

  if (!s || audio.paused) return;

  const offset = audio.playbackRate < 0.8 ? RATE_OFFSET.slow : RATE_OFFSET.fast;

  const elapsed = audio.currentTime - s.start - offset;

  const dur = s.end - s.start;

  if (elapsed < 0) return;

  const progress = Math.min(elapsed / dur, 1);

  const chars = document.querySelectorAll(".kc");

  if (!chars.length) return;

  const charProgress = progress * chars.length;

  const curIdx = Math.floor(charProgress);

  chars.forEach(function(c, i) {

    if (c.classList.contains("done")) return;

    if (i < curIdx) {

      c.classList.remove("active");

      c.classList.add("done");

    } else if (i === curIdx) {

      c.classList.add("active");

    }

  });

});


audio.addEventListener("seeking", function() {

  document.querySelectorAll(".kc").forEach(function(c) {

    c.classList.remove("done","active");

  });

});
